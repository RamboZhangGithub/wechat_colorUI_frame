# 业务介绍

目录结构

    ├── api                   // 接口封装
    │   ├── api                    // 接口
    │   ├── base64                 // 加密
    │   ├── connector              // 网络请求封装
    │   ├── env                    // 环境配置
    │   └── oss                    // 云上传封装
    ├── colorui                // colorui文件
    ├── components             // 组件
    │   ├── business               // 业务组件
    │   └── common                 // 通用组件
    ├── images                 // 图片文件
    │   ├── icon                   // 图标
    │   ├── imgs                   // 图片
    │   └── tabber                 // tab图标
    ├── pages                  // 页面文件
    ├── plugin                 // 插件文件
    ├── utils                  // 工具文件
    │   ├── imgs                   // 图片
    │   └── tabber                 // tab图标
    ├── app.js               // 小程序主入口
    ├── app.json             // 小程序页面配置
    └── project.config.json     // 小程序配置文件
